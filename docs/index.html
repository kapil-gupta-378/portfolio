check-palindrome.js 
// Two pointers: compare characters from both ends
function isPalindrome(s) {
  let left = 0, right = s.length - 1;
  while (left < right) {
    // If any pair mismatches, it's not a palindrome
    if (s[left] !== s[right]) return false;
    left++;
    right--;
  }
  // All pairs matched — it's a palindrome
  return true;
}
